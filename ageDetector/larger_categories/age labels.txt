def number_to_string(num):
    if num == 7:
        return "60+"
    elif num == 6:
        return "48-53"
    elif num == 5:
        return "38-43"
    elif num == 4:
        return "25-32"
    elif num == 3:
        return "15-20"
    elif num == 2:
        return "8-13"
    elif num == 1:
        return "4-6"
    else:
        return "0-2"
